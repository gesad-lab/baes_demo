        db.add(db_{entity_lower})
        db.commit()
        db.refresh(db_{entity_lower})
        return db_{entity_lower}
    except Exception as e:
        db.rollback()
        raise HTTPException(
            status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
            detail=f"Failed to create {Entity}: {{str(e)}}"
        )

        raise HTTPException(

@router.put("/{{{entity_name_lower}_id}}", response_model={Entity})

    """Update an existing {Entity}"""
    {entity_name_lower} = db.query({Entity}).filter({Entity}.id == {entity_name_lower}_id).first()
    if not {entity_name_lower}:

        return {entity_name_lower}

        db.rollback()
        raise HTTPException(
            status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
            detail=f"Failed to update {Entity}: {{str(e)}}"
        )

@router.delete("/{{{entity_name_lower}_id}}", status_code=status.HTTP_204_NO_CONTENT)
def delete_{entity_name_lower}({entity_name_lower}_id: int, db: Session = Depends(get_db)):

        db.commit()
